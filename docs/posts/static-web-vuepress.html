              # Tạo navbar

              chúng bổ sung thêm config nav trong file cấu hình như sau:

              module.exports = {
  themeConfig: {
    title: 'Tên blog',
    description: 'Mô tả về blog',
    // Thiết lập các liên kế trên navbar
    nav: [
      { text: 'Trang Chủ', link: '/' },
      { text: 'Giới Thiệu', link: '/about/' },
      { text: 'Ehkoo', link: 'https://ehkoo.com' },
    ],
  },
};

              # Tạo danh sách bài viết tự động

              Chúng ta tạo folder components trong .vuepress, tạo file all-posts.vue trong folder components

              |-.vuepress
| |-components
| | |-all-posts.vue
|-bai-viet
| |-bai-viet-1.md
|-package.json
|-README.md
              1
              2
              3
              4
              5
              6

              và thêm nội dụng sau vào file all-posts.vue

              // .vuepress/components/all-posts.vue
<template>
  <div>
    <p v-for="post in $site.pages.filter(p => p.path.includes('/posts'))">
      <router-link :to="post.path">{{ post.title }}</router-link>
    </p>
  </div>
</template>
``
